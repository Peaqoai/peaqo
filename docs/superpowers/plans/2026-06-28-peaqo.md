# Peaqo Multi-LLM Chat App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Turborepo-based multi-LLM chat SaaS with auth, onboarding, credit-gated chat, and an admin model-management panel.

**Architecture:** Single Next.js 15 app with Hono mounted at `app/api/[[...route]]/route.ts`. tRPC runs over Hono for all CRUD; chat streaming uses a plain Hono route consumed by AI SDK's `useChat`. Shared logic lives in `packages/*`. MongoDB via Mongoose. Auth via better-auth. Emails via React Email + Resend.

**Tech Stack:** Turborepo, pnpm, Next.js 15 (App Router), shadcn/ui, TanStack Query v5, Hono, tRPC v11, better-auth, Zod, Mongoose/MongoDB, React Email, Resend, Vercel AI SDK.

## Global Constraints

- Package manager: **pnpm** (workspace-based). Node ≥ 20.
- 1 credit = 1,000 tokens. Deduction: `ceil((tokensUsed / 1000) * model.creditMultiplier)`.
- Plans: `free | pro | ultimate | team`. Pro & Ultimate include team features.
- All LLM calls route through a per-model Cloudflare AI Gateway URL (from `Gateway` collection).
- Providers: `openai | anthropic | google | groq | openrouter | cloudflare`.
- Chat (`/app/chat`) is public to view; any action triggers the auth-gate modal.
- Stripe is out of scope — `plan`/`creditsLimit` set manually/mocked.

---

## Phases (execute in order; each phase yields working software)

- **Phase 1 — Foundation:** Turborepo + Next.js + shadcn + React Query + Hono + tRPC wired end-to-end (a working "hello tRPC" round-trip).
- **Phase 2 — Data layer:** MongoDB connection + all Mongoose models + Zod schemas.
- **Phase 3 — Auth + Email:** better-auth, Resend verification email, login/register pages.
- **Phase 4 — Onboarding:** multi-step profile/plan/team flow.
- **Phase 5 — Chat + Credits:** AI SDK streaming, guest auth-gate modal, credit deduction.
- **Phase 6 — Admin:** gateway + model management with live provider model fetch.

import mongoose from "mongoose";

let cached: Promise<typeof mongoose> | null = null;

export function connectDB(uri = process.env.MONGODB_URI): Promise<typeof mongoose> {
  if (!uri) throw new Error("MONGODB_URI is not set");
  if (!cached) cached = mongoose.connect(uri);
  return cached;
}

export function resetConnectionCache() { cached = null; } // ponytail: test-only reset
```

import mongoose, { Schema, model, models } from "mongoose";

const User = new Schema({
  email: { type: String, required: true, unique: true },
  name: { type: String, required: true },
  avatarUrl: String,
  plan: { type: String, enum: ["free","pro","ultimate","team"], default: "free" },
  role: { type: String, enum: ["user","admin"], default: "user" },
  orgId: { type: Schema.Types.ObjectId, ref: "Organisation" },
  onboardingComplete: { type: Boolean, default: false },
  creditsUsed: { type: Number, default: 0 },
  creditsLimit: { type: Number, default: 10 },
}, { timestamps: true });

const Organisation = new Schema({
  name: { type: String, required: true },
  ownerId: { type: Schema.Types.ObjectId, ref: "User", required: true },
  members: [{ type: Schema.Types.ObjectId, ref: "User" }],
}, { timestamps: true });

const Message = new Schema({
  role: { type: String, enum: ["user","assistant"], required: true },
  content: { type: String, required: true },
  tokenCount: { type: Number, default: 0 },
}, { _id: false, timestamps: true });

const Conversation = new Schema({
  userId: { type: Schema.Types.ObjectId, ref: "User", required: true },
  title: { type: String, default: "New chat" },
  modelId: { type: String, required: true },
  provider: { type: String, required: true },
  messages: [Message],
}, { timestamps: true });

const Gateway = new Schema({
  name: { type: String, required: true },
  url: { type: String, required: true },
}, { timestamps: true });

const ModelCfgSchema = new Schema({
  provider: { type: String, required: true },
  gatewayId: { type: Schema.Types.ObjectId, ref: "Gateway", required: true },
  modelId: { type: String, required: true },
  displayName: { type: String, required: true },
  creditMultiplier: { type: Number, default: 1 },
  enabled: { type: Boolean, default: true },
}, { timestamps: true });

export const UserModel = models.User ?? model("User", User);
export const OrgModel = models.Organisation ?? model("Organisation", Organisation);
export const ConversationModel = models.Conversation ?? model("Conversation", Conversation);
export const GatewayModel = models.Gateway ?? model("Gateway", Gateway);
export const ModelCfg = models.Model ?? model("Model", ModelCfgSchema);
```

import { createOpenAI } from "@ai-sdk/openai";
import { createAnthropic } from "@ai-sdk/anthropic";
import { createGoogleGenerativeAI } from "@ai-sdk/google";
import { createGroq } from "@ai-sdk/groq";
import { createOpenRouter } from "@openrouter/ai-sdk-provider";

export function creditsFor(tokensUsed: number, multiplier: number): number {
  return Math.ceil((tokensUsed / 1000) * multiplier);
}

export function resolveModel(opts: { provider: string; modelId: string; gatewayUrl: string }) {
  const { provider, modelId, gatewayUrl } = opts;
  switch (provider) {
    case "openai":     return createOpenAI({ baseURL: gatewayUrl })(modelId);
    case "anthropic":  return createAnthropic({ baseURL: gatewayUrl })(modelId);
    case "google":     return createGoogleGenerativeAI({ baseURL: gatewayUrl })(modelId);
    case "groq":       return createGroq({ baseURL: gatewayUrl })(modelId);
    case "openrouter": return createOpenRouter({ baseURL: gatewayUrl })(modelId);
    case "cloudflare": return createOpenAI({ baseURL: gatewayUrl })(modelId); // CF Workers AI is OpenAI-compatible
    default: throw new Error(`Unknown provider: ${provider}`);
  }
}

import { streamText } from "ai";
import { getSession } from "@peaqo/auth";
import { connectDB, UserModel, ConversationModel, ModelCfg, GatewayModel } from "@peaqo/db";
import { resolveModel, canAfford, nextCreditsUsed } from "@peaqo/trpc/llm/resolve";

export async function POST(req: Request) {
  const session = await getSession(req);
  if (!session) return new Response("Unauthorized", { status: 401 }); // guest -> client opens auth modal
  await connectDB();

  const { modelId, messages, conversationId } = await req.json();
  const user = await UserModel.findById(session.userId);
  if (!user || !canAfford(user)) return new Response("Out of credits", { status: 402 });

  const cfg = await ModelCfg.findOne({ modelId, enabled: true });
  if (!cfg) return new Response("Model not available", { status: 400 });
  const gateway = await GatewayModel.findById(cfg.gatewayId);

  const model = resolveModel({ provider: cfg.provider, modelId, gatewayUrl: gateway!.url });

  const result = streamText({
    model,
    messages,
    onFinish: async ({ usage }) => {
      const tokens = usage.totalTokens ?? 0;
      user.creditsUsed = nextCreditsUsed(user.creditsUsed, tokens, cfg.creditMultiplier);
      await user.save();
      // persist conversation + messages
      if (conversationId) {
        await ConversationModel.findByIdAndUpdate(conversationId, {
          $push: { messages: { $each: messages.slice(-1).map((m: any) => ({ role: m.role, content: m.content, tokenCount: tokens })) } },
        });
      }
    },
  });
  return result.toDataStreamResponse();
}

// Fetches the live model list per provider. Anthropic has no list API -> seeded.
const ANTHROPIC_SEED = ["claude-opus-4-8","claude-sonnet-4-6","claude-haiku-4-5-20251001"];
export async function listProviderModels(provider: string, apiKey: string): Promise<{ modelId: string }[]> {
  const urls: Record<string,string> = {
    openai: "https://api.openai.com/v1/models",
    google: "https://generativelanguage.googleapis.com/v1beta/models",
    groq: "https://api.groq.com/openai/v1/models",
    openrouter: "https://openrouter.ai/api/v1/models",
  };
  if (provider === "anthropic") return ANTHROPIC_SEED.map((modelId) => ({ modelId }));
  const url = urls[provider];
  if (!url) throw new Error(`No list endpoint for ${provider}`);
  const res = await fetch(url, { headers: { Authorization: `Bearer ${apiKey}` } });
  const json: any = await res.json();
  const items = json.data ?? json.models ?? [];
  return items.map((m: any) => ({ modelId: m.id ?? m.name }));
}
